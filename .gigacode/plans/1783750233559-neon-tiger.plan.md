   b. Выбрать приложение: "Mail"

4. **Создать облачную функцию**

**Файл `functions/index.js`:**
```javascript
const functions = require('@google-cloud/functions-framework');
const nodemailer = require('nodemailer');

functions.http('sendReportEmail', async (req, res) => {
    try {
        // Проверяем метод запроса
        if (req.method !== 'POST') {
            return res.status(405).json({ error: 'Method not allowed' });
        }
        
        const { htmlContent, recipientEmail, subject } = req.body;
        
        // Валидация входных данных
        if (!htmlContent || !recipientEmail) {
            return res.status(400).json({ 
                error: 'Missing required fields: htmlContent and recipientEmail' 
            });
        }
        
        // Настройка SMTP транспорта (Gmail с App Password)
        const transporter = nodemailer.createTransport({
            host: 'smtp.gmail.com',
            port: 587,
            secure: false,
            auth: {
                user: process.env.SMTP_USER,
                pass: process.env.SMTP_PASS
            }
        });
        
        // Отправка email
        const mailOptions = {
            from: process.env.SMTP_FROM || '"Ohrana" <' + process.env.SMTP_USER + '>',
            to: recipientEmail,
            subject: subject || 'Отчет о смене - Ohrana Security System',
            html: htmlContent,
            attachments: [{
                filename: 'shift_report.html',
                content: htmlContent,
                contentType: 'text/html'
            }]
        };
        
        const info = await transporter.sendMail(mailOptions);
        
        res.status(200).json({ 
            success: true, 
            messageId: info.messageId,
            message: 'Email успешно отправлен',
            to: recipientEmail
        });
        
    } catch (error) {
        console.error('Error sending email:', error);
        res.status(500).json({ 
            success: false, 
            error: error.message 
        });
    }
});
```

**Файл `functions/package.json` (обновить dependencies):**
```json

    "@google-cloud/functions-framework": "^3.0.0",

```

**Добавить новый метод:**

```kotlin
/**
 * Отправляет HTML отчет на почту НАПРЯМУЮ (без Яндекс.Диска)
 * Генерирует HTML и отправляет его в Firebase Cloud Function
 * @param shiftId ID смены
 * @param shiftDatabase Менеджер базы данных смен
 * @param recipientEmail Email получателя
 * @param subject Тема письма (опционально)
 * @return Result с результатом отправки
 */
fun sendHtmlReportDirectToEmail(
    shiftId: String,
    shiftDatabase: ShiftDatabaseManager,
    recipientEmail: String,
    subject: String? = null
): Result<String> {
    return try {
        // 1. Генерируем HTML отчет
        val htmlContent = generateHtmlReport(shiftId, shiftDatabase)
        if (htmlContent == null) {
            return Result.failure(Exception("Ошибка генерации HTML отчета"))
        }
        
        Log.i(TAG, "sendHtmlReportDirectToEmail: HTML generated, size=${htmlContent.length} bytes")
        
        // 2. Вызываем Firebase Cloud Function
        // Замени на реальный URL твоей функции после deploy
        val functionUrl = "https://your-region-your-project.cloudfunctions.net/sendReportEmail"
        val client = okhttp3.OkHttpClient()
        
        // Формируем JSON-тело запроса
        val jsonBody = """{"htmlContent":${"$"}htmlContent,"recipientEmail":${"$"}recipientEmail${if (subject != null) ""","subject":"${"$"}subject"" else ""}}"""
        val mediaType = "application/json; charset=utf-8".toMediaType()
        val requestBody = okhttp3.RequestBody.create(mediaType, jsonBody)
        
        val request = okhttp3.Request.Builder()
            .url(functionUrl)
            .post(requestBody)
            .addHeader("Content-Type", "application/json")
            .build()
        
        val response = client.newCall(request).execute()
        
        if (!response.isSuccessful) {
            val errorMessage = response.body?.string() ?: "Unknown error"
            Log.e(TAG, "Firebase Function returned code: ${response.code}, response: $errorMessage")
            return Result.failure(Exception("Firebase Function error: ${response.code} ${response.message}"))
        }
        
        val responseBody = response.body?.string() ?: ""
        
        Log.i(TAG, "sendHtmlReportDirectToEmail: Success, response=$responseBody")
        
        // Возвращаем сообщение об успехе
        val successMessage = "Отчет отправлен на $recipientEmail"
        Result.success(successMessage)
        
    } catch (e: Exception) {
        val errorMsg = e.message ?: "Неизвестная ошибка"
        Log.e(TAG, "sendHtmlReportDirectToEmail: Error: $errorMsg", e)
        Result.failure(Exception("Ошибка отправки отчета: $errorMsg"))
    }
}
```

**ОПЦИОНАЛЬНО:** Можно сжать HTML перед отправкой (если будете часто превышать лимиты):

```kotlin
/**
 * Сжимает HTML в base64 (для отправки больших файлов)
 */
fun compressHtmlToBase64(htmlContent: String): String {
    return android.util.Base64.encodeToString(
        htmlContent.toByteArray(Charsets.UTF_8),
        android.util.Base64.NO_WRAP
    )
}
```

### Важные замечания:

1. **Gmail App Password:**
   - Для Gmail нужно использовать "App Password" (не обычный пароль)
   - Настройка через: https://myaccount.google.com/apppasswords
   - Выбрать приложение: "Mail", устройство: "Other (Custom name)"

2. **Firebase Environment Variables:**
   ```bash
   firebase functions:config:set \
     smtp.user="твой-email@gmail.com" \
     smtp.pass="твой-app-password-без-пробелов" \
     smtp.from="Ohrana Security System <твой-email@gmail.com>"
   ```

3. **Размер HTML-файла:**
   - Firebase Functions имеет лимит на размер запроса (~10MB)
